025    public static void refreshIndex(Client client,String index){
026        client.admin().indices().refresh(new RefreshRequest(index)).actionGet();
027    }

029    public static void createIndexIfNotExist(Client client,String index){
030        // Creation of working index
031        if(client.admin().indices().prepareExists(index).execute().actionGet().exists() == false){
032            client.admin().indices().prepareCreate(index).execute().actionGet();
033        }
034    }
035
036    public static void deleteDocumentsInIndex(Client client,String index){
037        client.prepareDeleteByQuery(index).setQuery(QueryBuilders.matchAllQuery()).execute().actionGet();
038    }
039
040    public static void deleteDocumentInIndex(Client client,String index, String type,String id){
041        client.prepareDelete(index, type, id).execute().actionGet();
042    }

084    /**
085     * Create a test BulkOperation with ES in memory
086     * @param logger
087     * @return
088     */
089    public static BulkOperation getMemoryBulkOperation(ESLogger logger){
090        Settings settings = ImmutableSettings.settingsBuilder()
091                .put("gateway.type","none")
092                .put("index.gateway.type", "none")
093                .put("index.store.type", "memory")
094                .put("path.data","target/data")
095                .build();
096
097        Node node = NodeBuilder.nodeBuilder().local(true).settings(settings).node();
098        final Client client = node.client();
099
100
101        return new BulkOperation(client,logger){
102            @Override
103            public BulkOperation setIndex(String index) {
104                super.setIndex(index);
105                // We create the index in memory if it does'nt exist
106                if(!client.admin().indices().prepareExists(index).execute().actionGet().exists()){
107                    client.admin().indices().prepareCreate(index).execute().actionGet();
108                }
109                return this;
110            }
111
112            @Override
113            public void create(String index, String type, String id, long version, XContentBuilder builder) {
114                try{
115                    logger.info(" CREATE : " + this.index + "/" + this.type + "/" + id + " => " + builder.string());
116                    client.prepareIndex(this.index, this.type).setSource(builder).execute().actionGet();
117                }catch(Exception e){
118                    logger.error("Error generating CREATE");
119                }
120            }
121
122            @Override
123            public void index(String index, String type, String id, long version, XContentBuilder builder) {
124                try{
125                    logger.info(" INDEX: " + this.index + "/" + this.type + "/" + id + " => " + builder.string());
126                    IndexResponse r = client.prepareIndex(this.index,this.type,id).setSource(builder).execute().actionGet();
127                }catch(Exception e){
128                    logger.error("Error generating INDEX");
129                }
130            }
131
132            @Override
133            public void delete(String index, String type, String id) {
134                logger.info(" DELETE: " + this.index + "/" + this.type + "/" + id);
135                client.prepareDelete(this.index, this.type, id).execute().actionGet();
136            }
137        };
138    }

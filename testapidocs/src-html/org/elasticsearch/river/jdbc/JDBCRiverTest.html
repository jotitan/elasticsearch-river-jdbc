165    @Test
166    public void testComplexMergerInMemory()throws Exception{
167        BulkOperation op = ESUtilTest.getMemoryBulkOperation(logger).setIndex(INDEX_NAME).setType("car");
168        ESUtilTest.createIndexIfNotExist(op.getClient(),ESUtilTest.NAME_INDEX_RIVER);
169
170        RiverSettings settings = createSettings();
171        JDBCRiver river = new JDBCRiver(new RiverName(ESUtilTest.TYPE_INDEX_RIVER,ESUtilTest.TYPE_INDEX_RIVER),settings,ESUtilTest.NAME_INDEX_RIVER,op.getClient());
172        river.delay = false;
173        river.riverStrategy.run();
174        ESUtilTest.refreshIndex(op.getClient(), INDEX_NAME);
175        Assert.assertEquals(op.getClient().prepareSearch(INDEX_NAME).execute().actionGet().getHits().getTotalHits(), 5);
176
177        // Add new entry to test if it's indexed
178        Connection conn = SQLUtilTest.createConnection();
179        SQLUtilTest.addDataTest(conn, 6, "car6", "2012-07-02 14:00:00", new Integer[]{3});
180        conn.close();
181
182
183        river.riverStrategy.run();
184        ESUtilTest.refreshIndex(op.getClient(), INDEX_NAME);
185        Assert.assertEquals(op.getClient().prepareSearch(INDEX_NAME).execute().actionGet().getHits().getTotalHits(), 6);
186
187        // Add a new object the same last modification date. We check it's really index
188        conn = SQLUtilTest.createConnection();
189        SQLUtilTest.addDataTest(conn, 7, "car7", "2012-07-02 14:00:00", new Integer[]{1});
190        conn.close();
191
192        river.riverStrategy.run();
193        ESUtilTest.refreshIndex(op.getClient(), INDEX_NAME);
194        Assert.assertEquals(op.getClient().prepareSearch(INDEX_NAME).execute().actionGet().getHits().getTotalHits(), 7);
195
196        // Add a new object the same last modification date. It must be avoid
197        conn = SQLUtilTest.createConnection();
198        SQLUtilTest.addDataTest(conn, 8, "car8", "2012-07-02 14:00:00", new Integer[]{2,4});
199        conn.close();
200
201        river.riverStrategy.run();
202        ESUtilTest.refreshIndex(op.getClient(), INDEX_NAME);
203        Assert.assertEquals(op.getClient().prepareSearch(INDEX_NAME).execute().actionGet().getHits().getTotalHits(), 7);
204
205    }

249    public void testDuplicateJoin()throws Exception{
250        Connection connection = SQLUtilTest.createConnection();
251        SQLUtilTest.truncateTable(connection,"car");
252        SQLUtilTest.truncateTable(connection,"car_opt_have");
253        SQLUtilTest.truncateTable(connection,"car_color_have");
254
255        // Add a car with two colors and only one option
256        SQLUtilTest.addDataTest(connection, 1,"car1","2012-07-12 12:00:00",new Integer[]{1},new Integer[]{2,3});
257        connection.close();
258
259        BulkOperation op = ESUtilTest.getMemoryBulkOperation(logger).setIndex(INDEX_NAME).setType("car");
260        ESUtilTest.createIndexIfNotExist(op.getClient(), ESUtilTest.NAME_INDEX_RIVER);
261
262        RiverSettings settings = createSettings();
263        ((Map<String,Object>)settings.settings().get("jdbc")).put("sql",manyJoinQuery);
264        ((Map<String,Object>)settings.settings().get("jdbc")).put("mapping",mappingManyJoinQuery);
265
266        JDBCRiver river = new JDBCRiver(new RiverName(ESUtilTest.TYPE_INDEX_RIVER,ESUtilTest.TYPE_INDEX_RIVER),settings,ESUtilTest.NAME_INDEX_RIVER,op.getClient());
267        river.delay = false;
268        river.riverStrategy.run();
269
270        ESUtilTest.refreshIndex(op.getClient(), INDEX_NAME);
271        SearchResponse response = op.getClient().prepareSearch(INDEX_NAME).execute().actionGet();
272        Assert.assertEquals(response.getHits().getTotalHits(), 1);
273        Assert.assertEquals(((List<Object>)response.getHits().hits()[0].sourceAsMap().get("options")).size(),1);
274        Assert.assertEquals(((List<Object>)response.getHits().hits()[0].sourceAsMap().get("colors")).size(),2);
275    }

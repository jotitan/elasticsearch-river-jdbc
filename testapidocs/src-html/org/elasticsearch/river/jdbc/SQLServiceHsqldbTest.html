103    @Test
104    public void testMerger()throws Exception{
105        String query = "select 'index' as \"_operation\", id as _id, id, label, o.label " +
106                "            from car left join car_opt_have co on car.id = co.id_car" +
107                "            left join opt o on o.id = co.id_opt";
108        List<Object> mapping = ESUtilTest.createMapping("_operation","_id","car.id","car.label","car.options");
109        Connection connection = sqlService.getConnection("org.hsqldb.jdbcDriver","jdbc:hsqldb:mem:test", "SA", "",true);
110        PreparedStatement ps = sqlService.prepareStatement(connection,query);
111        sqlService.treat(ps,5,"index",ESUtilTest.getMockBulkOperation(logger).setIndex(INDEX_NAME).setType("car"),mapping);
112    }

123    @Test
124    public void testComplexMergerInMemory()throws Exception{
125        Connection connection = sqlService.getConnection("org.hsqldb.jdbcDriver","jdbc:hsqldb:mem:test", "SA", "",true);
126        PreparedStatement ps = sqlService.prepareStatement(connection,complexQuery2);
127        BulkOperation op = ESUtilTest.getMemoryBulkOperation(logger).setIndex(INDEX_NAME).setType("car");
128
129        sqlService.treat(ps,5,"index",op,mappingComplexQuery2);
130        refreshIndex(op.getClient(),INDEX_NAME);
131        Assert.assertEquals(op.getClient().prepareSearch(INDEX_NAME).execute().actionGet().getHits().getTotalHits(), 5);
132
133        // Add new entry to test if it's indexed
134        Connection conn = SQLUtilTest.createConnection();
135        SQLUtilTest.addDataTest(conn, 6, "car6", "2012-07-02 14:00:00", new Integer[]{3});
136        conn.close();
137
138        sqlService.treat(ps,5,"index",op,mappingComplexQuery2);
139        refreshIndex(op.getClient(),INDEX_NAME);
140        Assert.assertEquals(op.getClient().prepareSearch(INDEX_NAME).execute().actionGet().getHits().getTotalHits(), 6);
141    }

066    public SQLService setRounding(String rounding) {
067        if ("ceiling".equalsIgnoreCase(rounding)) {
068            this.rounding = BigDecimal.ROUND_CEILING;
069        } else if ("down".equalsIgnoreCase(rounding)) {
070            this.rounding = BigDecimal.ROUND_DOWN;
071        } else if ("floor".equalsIgnoreCase(rounding)) {
072            this.rounding = BigDecimal.ROUND_FLOOR;
073        } else if ("halfdown".equalsIgnoreCase(rounding)) {
074            this.rounding = BigDecimal.ROUND_HALF_DOWN;
075        } else if ("halfeven".equalsIgnoreCase(rounding)) {
076            this.rounding = BigDecimal.ROUND_HALF_EVEN;
077        } else if ("halfup".equalsIgnoreCase(rounding)) {
078            this.rounding = BigDecimal.ROUND_HALF_UP;
079        } else if ("unnecessary".equalsIgnoreCase(rounding)) {
080            this.rounding = BigDecimal.ROUND_UNNECESSARY;
081        } else if ("up".equalsIgnoreCase(rounding)) {
082            this.rounding = BigDecimal.ROUND_UP;
083        }
084        return this;
085    }

126    /**
127     * Prepare statement for processing a river sync table
128     *
129     * @param connection the JDBC connection
130     * @param riverName the name of the river
131     * @param optype the operation type
132     * @param interval the interval to cover, in milliseconds (back from current
133     * time)
134     * @return a preapred river table statement
135     * @throws SQLException
136     */
137    public PreparedStatement prepareRiverTableStatement(Connection connection, String riverName, String optype, long interval)
138            throws SQLException {
139        PreparedStatement p = connection.prepareStatement("select * from " + riverName + " where source_operation = ? and target_operation = 'n/a' and source_timestamp between ? and ?",
140                ResultSet.TYPE_FORWARD_ONLY, ResultSet.CONCUR_READ_ONLY);
141        p.setString(1, optype);
142        java.util.Date d = new java.util.Date();
143        long now = d.getTime();
144        p.setTimestamp(2, new Timestamp(now - interval));
145        p.setTimestamp(3, new Timestamp(now));
146        return p;
147    }

179    /**
180     * Index data returns by the request in statement. At the end of treatment, return the modification date of the last indexed object. When river come again, treatment start a this point.
181     * @param statement Statement which contain request
182     * @param fetchsize  Number of elements to index
183     * @param defaultOperation Default operation if nothing is define in select results
184     * @param aliasDateField
185     * @return the date of modification of the last indexed document. The results are not sorted, the date is getting in comparing all date ang keep the max
186     * @throws SQLException
187     */
188    public String treat(PreparedStatement statement, int fetchsize, String defaultOperation,BulkOperation bulkOp, List<Object> mappings) throws SQLException,IOException {
189        long beginDate = System.currentTimeMillis();
190        ResultSet results = getResultsWithBounds(statement,fetchsize);
191        long endDate = System.currentTimeMillis();
192
193        logger.info("Request time : " + ((endDate-beginDate)/1000) + "s");
194        /* Treat all elements. When id change, we flush. If an id is not read completly, we load the next elements */
195        ComplexMerger merger = new ComplexMerger();
196       
197        String currentId = null;
198        String currentOperation = defaultOperation;
199        ResultSetMetaData metadata = results.getMetaData();
200
201        String currentModificationDate = null;  // modification date of last indexed element, search the max
202
203        int pos = 0;
204        while(results.next()){
205           pos++;
206            String id = results.getString("_id");
207            /* If id is different, save the datas and init the contexte */
208            if(currentId!=null && !currentId.equals(id)){
209                saveIndex(merger.getRoot(),bulkOp);
210                merger.reset();
211               }
212            currentId = id;
213            if(defaultOperation == null){
214                currentOperation = results.getString("_operation");
215            }
216            //logger.info("Treat document with id : " + id + " => " + ((defaultOperation == null)?currentOperation:defaultOperation));
217
218            merger.getRoot().setOperation(currentOperation);
219            merger.getRoot().setId(currentId);
220            
221            List<String> keys = new ArrayList<String>();
222            List<Object> values = new ArrayList<Object>();
223
224            for(int i = 1 ; i <= metadata.getColumnCount() ; i++){
225                String columnName = (String)mappings.get(i-1);
226                if(!columnName.startsWith("_")){
227                    keys.add(columnName);
228                    values.add(parseType(results,i,metadata.getColumnType(i),columnName,metadata.getColumnCount()));
229                }
230                /* Save the modification date of document */
231                if(columnName.toLowerCase().equals(JDBCRiver.FIELD_MODIFICATION_DATE)){
232                    Object date = parseType(results,i,93,columnName,metadata.getColumnCount());
233                    if(date!=null){
234                        String formatDate = DateUtil.formatDateStandard(DateUtil.parseDateISO(date.toString()));
235                        if(formatDate!=null){
236                            if(currentModificationDate == null || formatDate.compareTo(currentModificationDate) > 0){
237                                currentModificationDate  = formatDate;
238                            }
239                        }
240                    }
241                }
242            }
243            merger.row(keys,values);
244        }
245        if(pos == 0){
246            return null;
247        }
248
249        saveIndex(merger.getRoot(),bulkOp);
250        bulkOp.flush();
251        results.close();
252        return currentModificationDate;
253    }

255    private void saveIndex(ComplexMerger.PropertyRoot root,BulkOperation bulkOp)throws IOException{
256        if(root.getOperation().equals("create")){
257            bulkOp.create(null,null,root.getId(),0L,root.getXBuilder());
258        }
259        if(root.getOperation().equals("index")){
260            bulkOp.index(null,null,root.getId(),0L,root.getXBuilder());
261        }
262        if(root.getOperation().equals("delete")){
263            bulkOp.delete(null,null,root.getId());
264        }
265    }

305    /**
306     * Get next row from a river table. The river table contains embedded SQL
307     * which is executed.
308     *
309     * @param result the result set
310     * @param listener the row listener
311     * @return true if there was a row being executed, false otherwise
312     * @throws SQLException
313     * @throws IOException
314     */
315    public boolean nextRiverTableRow(ResultSet result, RowListener listener)
316            throws SQLException, IOException {
317        if (result.next()) {
318            ResultSetMetaData metadata = result.getMetaData();
319            String operation = null;
320            String index = null;
321            String type = null;
322            String id = null;
323            String sql = null;
324            int columns = metadata.getColumnCount();
325            for (int i = 1; i <= columns; i++) {
326                String name = metadata.getColumnLabel(i);
327                if ("_index".equalsIgnoreCase(name)) {
328                    index = result.getString(i);
329                } else if ("_type".equalsIgnoreCase(name)) {
330                    type = result.getString(i);
331                } else if ("_id".equalsIgnoreCase(name)) {
332                    id = result.getString(i);
333                } else if ("source_operation".equalsIgnoreCase(name)) {
334                    operation = result.getString(i);
335                } else if ("source_sql".equalsIgnoreCase(name)) {
336                    sql = result.getString(i);
337                }
338            }
339            if (sql != null) {
340                // execute embedded SQL
341                PreparedStatement stmt = prepareStatement(connection, sql);
342                ResultSet rs = stmt.executeQuery();
343                long rows = 0L;
344                if (rs.next()) {
345                    processRow(rs, listener, operation, index, type, id);
346                    rows++;
347                }
348                logger.info("embedded sql gave " + rows + " rows");
349                rs.close();
350                stmt.close();
351            }
352            return true;
353        }
354        return false;
355    }

357    /**
358     * Parse of value of resultset with the good type
359     * @param result
360     * @param i
361     * @param metadata
362     * @param name
363     * @return The parse value
364     * @throws SQLException
365     * @throws IOException
366     */
367    private Object parseType(ResultSet result,Integer i,int type,String name,int columnCount)throws SQLException,IOException{
368
369        switch (type) {
370            /**
371             * The JDBC types CHAR, VARCHAR, and LONGVARCHAR are closely
372             * related. CHAR represents a small, fixed-length character
373             * string, VARCHAR represents a small, variable-length character
374             * string, and LONGVARCHAR represents a large, variable-length
375             * character string.
376             */
377            case Types.CHAR:
378            case Types.VARCHAR:
379            case Types.LONGVARCHAR: {
380                return result.getString(i);
381            }
382            case Types.NCHAR:
383            case Types.NVARCHAR:
384            case Types.LONGNVARCHAR: {
385                return result.getNString(i);
386            }
387            /**
388             * The JDBC types BINARY, VARBINARY, and LONGVARBINARY are
389             * closely related. BINARY represents a small, fixed-length
390             * binary value, VARBINARY represents a small, variable-length
391             * binary value, and LONGVARBINARY represents a large,
392             * variable-length binary value
393             */
394            case Types.BINARY:
395            case Types.VARBINARY:
396            case Types.LONGVARBINARY: {
397               return result.getBytes(name);
398            }
399            /**
400             * The JDBC type ARRAY represents the SQL3 type ARRAY.
401             *
402             * An ARRAY value is mapped to an instance of the Array
403             * interface in the Java programming language. If a driver
404             * follows the standard implementation, an Array object
405             * logically points to an ARRAY value on the server rather than
406             * containing the elements of the ARRAY object, which can
407             * greatly increase efficiency. The Array interface contains
408             * methods for materializing the elements of the ARRAY object on
409             * the client in the form of either an array or a ResultSet
410             * object.
411             */
412            case Types.ARRAY: {
413                Array a = result.getArray(i);
414                return a != null ? a.toString() : null;
415            }
416            /**
417             * The JDBC type BIGINT represents a 64-bit signed integer value
418             * between -9223372036854775808 and 9223372036854775807.
419             *
420             * The corresponding SQL type BIGINT is a nonstandard extension
421             * to SQL. In practice the SQL BIGINT type is not yet currently
422             * implemented by any of the major databases, and we recommend
423             * that its use be avoided in code that is intended to be
424             * portable.
425             *
426             * The recommended Java mapping for the BIGINT type is as a Java
427             * long.
428             */
429            case Types.BIGINT: {
430                return result.getLong(i);
431            }
432            /**
433             * The JDBC type BIT represents a single bit value that can be
434             * zero or one.
435             *
436             * SQL-92 defines an SQL BIT type. However, unlike the JDBC BIT
437             * type, this SQL-92 BIT type can be used as a parameterized
438             * type to define a fixed-length binary string. Fortunately,
439             * SQL-92 also permits the use of the simple non-parameterized
440             * BIT type to represent a single binary digit, and this usage
441             * corresponds to the JDBC BIT type. Unfortunately, the SQL-92
442             * BIT type is only required in "full" SQL-92 and is currently
443             * supported by only a subset of the major databases. Portable
444             * code may therefore prefer to use the JDBC SMALLINT type,
445             * which is widely supported.
446             */
447            case Types.BIT: {
448                return result.getInt(i);
449            }
450            /**
451             * The JDBC type BOOLEAN, which is new in the JDBC 3.0 API, maps
452             * to a boolean in the Java programming language. It provides a
453             * representation of true and false, and therefore is a better
454             * match than the JDBC type BIT, which is either 1 or 0.
455             */
456            case Types.BOOLEAN: {
457                return result.getBoolean(i);
458            }
459            /**
460             * The JDBC type BLOB represents an SQL3 BLOB (Binary Large
461             * Object).
462             *
463             * A JDBC BLOB value is mapped to an instance of the Blob
464             * interface in the Java programming language. If a driver
465             * follows the standard implementation, a Blob object logically
466             * points to the BLOB value on the server rather than containing
467             * its binary data, greatly improving efficiency. The Blob
468             * interface provides methods for materializing the BLOB data on
469             * the client when that is desired.
470             */
471            case Types.BLOB: {
472                Blob blob = result.getBlob(i);
473                if (blob != null) {
474                    long n = blob.length();
475                    if (n > Integer.MAX_VALUE) {
476                        throw new IOException("can't process blob larger than Integer.MAX_VALUE");
477                    }
478                    byte[] tab = blob.getBytes(1, (int)n);
479                    blob.free();
480                    return tab;
481                }
482                break;
483            }
484            /**
485             * The JDBC type CLOB represents the SQL3 type CLOB (Character
486             * Large Object).
487             *
488             * A JDBC CLOB value is mapped to an instance of the Clob
489             * interface in the Java programming language. If a driver
490             * follows the standard implementation, a Clob object logically
491             * points to the CLOB value on the server rather than containing
492             * its character data, greatly improving efficiency. Two of the
493             * methods on the Clob interface materialize the data of a CLOB
494             * object on the client.
495             */
496            case Types.CLOB: {
497                Clob clob = result.getClob(i);
498                if (clob != null) {
499                    long n = clob.length();
500                    if (n > Integer.MAX_VALUE) {
501                        throw new IOException("can't process clob larger than Integer.MAX_VALUE");
502                    }
503                    String str = clob.getSubString(1, (int)n);
504                    clob.free();
505                    return str;
506                }
507                break;
508            }
509            case Types.NCLOB: {
510                NClob nclob = result.getNClob(i);
511                if (nclob != null) {
512                    long n = nclob.length();
513                    if (n > Integer.MAX_VALUE) {
514                        throw new IOException("can't process nclob larger than Integer.MAX_VALUE");
515                    }
516                    String str = nclob.getSubString(1, (int)n);
517                    nclob.free();
518                    return str;
519                }
520                break;
521            }
522            /**
523             * The JDBC type DATALINK, new in the JDBC 3.0 API, is a column
524             * value that references a file that is outside of a data source
525             * but is managed by the data source. It maps to the Java type
526             * java.net.URL and provides a way to manage external files. For
527             * instance, if the data source is a DBMS, the concurrency
528             * controls it enforces on its own data can be applied to the
529             * external file as well.
530             *
531             * A DATALINK value is retrieved from a ResultSet object with
532             * the ResultSet methods getURL or getObject. If the Java
533             * platform does not support the type of URL returned by getURL
534             * or getObject, a DATALINK value can be retrieved as a String
535             * object with the method getString.
536             *
537             * java.net.URL values are stored in a database using the method
538             * setURL. If the Java platform does not support the type of URL
539             * being set, the method setString can be used instead.
540             *
541             *
542             */
543            case Types.DATALINK: {
544                return result.getURL(i);
545            }
546            /**
547             * The JDBC DATE type represents a date consisting of day,
548             * month, and year. The corresponding SQL DATE type is defined
549             * in SQL-92, but it is implemented by only a subset of the
550             * major databases. Some databases offer alternative SQL types
551             * that support similar semantics.
552             */
553            case Types.DATE: {
554                try {
555                    Date d = result.getDate(i);
556                    return d != null ? DateUtil.formatDateISO(d.getTime()) : null;
557                } catch (SQLException e) {
558                    return null;
559                }
560            }
561            case Types.TIME: {
562                try {
563                    Time t = result.getTime(i);
564                   return t != null ? DateUtil.formatDateISO(t.getTime()) : null;
565                } catch (SQLException e) {
566                    return null;
567                }
568            }
569            case Types.TIMESTAMP: {
570                try {
571                    Timestamp t = result.getTimestamp(i);
572                    return t != null ? DateUtil.formatDateISO(t.getTime()) : null;
573                } catch (SQLException e) {
574                    // java.sql.SQLException: Cannot convert value '0000-00-00 00:00:00' from column ... to TIMESTAMP.
575                    return null;
576                }
577            }
578            /**
579             * The JDBC types DECIMAL and NUMERIC are very similar. They
580             * both represent fixed-precision decimal values.
581             *
582             * The corresponding SQL types DECIMAL and NUMERIC are defined
583             * in SQL-92 and are very widely implemented. These SQL types
584             * take precision and scale parameters. The precision is the
585             * total number of decimal digits supported, and the scale is
586             * the number of decimal digits after the decimal point. For
587             * most DBMSs, the scale is less than or equal to the precision.
588             * So for example, the value "12.345" has a precision of 5 and a
589             * scale of 3, and the value ".11" has a precision of 2 and a
590             * scale of 2. JDBC requires that all DECIMAL and NUMERIC types
591             * support both a precision and a scale of at least 15.
592             *
593             * The sole distinction between DECIMAL and NUMERIC is that the
594             * SQL-92 specification requires that NUMERIC types be
595             * represented with exactly the specified precision, whereas for
596             * DECIMAL types, it allows an implementation to add additional
597             * precision beyond that specified when the type was created.
598             * Thus a column created with type NUMERIC(12,4) will always be
599             * represented with exactly 12 digits, whereas a column created
600             * with type DECIMAL(12,4) might be represented by some larger
601             * number of digits.
602             *
603             * The recommended Java mapping for the DECIMAL and NUMERIC
604             * types is java.math.BigDecimal. The java.math.BigDecimal type
605             * provides math operations to allow BigDecimal types to be
606             * added, subtracted, multiplied, and divided with other
607             * BigDecimal types, with integer types, and with floating point
608             * types.
609             *
610             * The method recommended for retrieving DECIMAL and NUMERIC
611             * values is ResultSet.getBigDecimal. JDBC also allows access to
612             * these SQL types as simple Strings or arrays of char. Thus,
613             * Java programmers can use getString to receive a DECIMAL or
614             * NUMERIC result. However, this makes the common case where
615             * DECIMAL or NUMERIC are used for currency values rather
616             * awkward, since it means that application writers have to
617             * perform math on strings. It is also possible to retrieve
618             * these SQL types as any of the Java numeric types.
619             */
620            case Types.DECIMAL:
621            case Types.NUMERIC: {
622                BigDecimal bd = result.getBigDecimal(i);
623                return bd != null ? bd.toString() : null;
624            }
625            /**
626             * The JDBC type DOUBLE represents a "double precision" floating
627             * point number that supports 15 digits of mantissa.
628             *
629             * The corresponding SQL type is DOUBLE PRECISION, which is
630             * defined in SQL-92 and is widely supported by the major
631             * databases. The SQL-92 standard leaves the precision of DOUBLE
632             * PRECISION up to the implementation, but in practice all the
633             * major databases supporting DOUBLE PRECISION support a
634             * mantissa precision of at least 15 digits.
635             *
636             * The recommended Java mapping for the DOUBLE type is as a Java
637             * double.
638             */
639            case Types.DOUBLE: {
640                return result.getDouble(i);
641            }
642            /**
643             * The JDBC type FLOAT is basically equivalent to the JDBC type
644             * DOUBLE. We provided both FLOAT and DOUBLE in a possibly
645             * misguided attempt at consistency with previous database APIs.
646             * FLOAT represents a "double precision" floating point number
647             * that supports 15 digits of mantissa.
648             *
649             * The corresponding SQL type FLOAT is defined in SQL-92. The
650             * SQL-92 standard leaves the precision of FLOAT up to the
651             * implementation, but in practice all the major databases
652             * supporting FLOAT support a mantissa precision of at least 15
653             * digits.
654             *
655             * The recommended Java mapping for the FLOAT type is as a Java
656             * double. However, because of the potential confusion between
657             * the double precision SQL FLOAT and the single precision Java
658             * float, we recommend that JDBC programmers should normally use
659             * the JDBC DOUBLE type in preference to FLOAT.
660             */
661            case Types.FLOAT: {
662                return result.getDouble(i);
663            }
664            /**
665             * The JDBC type INTEGER represents a 32-bit signed integer
666             * value ranging between -2147483648 and 2147483647.
667             *
668             * The corresponding SQL type, INTEGER, is defined in SQL-92 and
669             * is widely supported by all the major databases. The SQL-92
670             * standard leaves the precision of INTEGER up to the
671             * implementation, but in practice all the major databases
672             * support at least 32 bits.
673             *
674             * The recommended Java mapping for the INTEGER type is as a
675             * Java int.

823    private void processRow(ResultSet result, RowListener listener, String operation, String index, String type, String id)
824            throws SQLException, IOException {
825        LinkedList<String> keys = new LinkedList();
826        LinkedList<Object> values = new LinkedList();
827        ResultSetMetaData metadata = result.getMetaData();
828        int columns = metadata.getColumnCount();
829        for (int i = 1; i <= columns; i++) {
830            String name = metadata.getColumnLabel(i);
831            if ("_index".equalsIgnoreCase(name)) {
832                index = result.getString(i);
833                continue;
834            } else if ("_type".equalsIgnoreCase(name)) {
835                type = result.getString(i);
836                continue;
837            } else if ("_id".equalsIgnoreCase(name)) {
838                id = result.getString(i);
839                continue;
840            }
841            keys.add(name);
842            values.add(parseType(result,i,metadata.getColumnType(i),name,metadata.getColumnCount()));
843        }
844        if (id == null) {
845            id = Integer.toString(result.getRow());
846        }
847        if (listener != null) {
848            listener.row(operation, index, type, id, keys, values);
849        }
850    }

882    private void bind(PreparedStatement pstmt, int i, Object value) throws SQLException {
883        if (value == null) {
884            pstmt.setNull(i, Types.VARCHAR);
885        } else if (value instanceof String) {
886            String s = (String) value;
887            if ("$now".equals(s)) {
888                pstmt.setDate(i, new Date(new java.util.Date().getTime()));
889            } else {
890                pstmt.setString(i, (String) value);
891            }
892        } else if (value instanceof Integer) {
893            pstmt.setInt(i, ((Integer) value).intValue());
894        } else if (value instanceof Long) {
895            pstmt.setLong(i, ((Long) value).longValue());
896        } else if (value instanceof BigDecimal) {
897            pstmt.setBigDecimal(i, (BigDecimal) value);
898        } else if (value instanceof Date) {
899            pstmt.setDate(i, (Date) value);
900        } else if (value instanceof Timestamp) {
901            pstmt.setTimestamp(i, (Timestamp) value);
902        } else if (value instanceof Float) {
903            pstmt.setFloat(i, ((Float) value).floatValue());
904        } else if (value instanceof Double) {
905            pstmt.setDouble(i, ((Double) value).doubleValue());
906        } else {
907            pstmt.setObject(i, value);
908        }
909    }

911    /**
912     * Acknowledge a bulk item response back to the river table. Fill columns
913     * target_timestamp, taget_operation, target_failed, target_message.
914     *
915     * @param riverName
916     * @param response
917     * @throws IOException
918     */
919    @Override
920    public void acknowledge(String riverName, BulkItemResponse[] response) throws IOException {
921        if (response == null) {
922            logger.warn("can't acknowledge null bulk response");
923        }
924        try {
925            String sql = " update " + riverName + " set target_timestamp = ?, target_operation = ?, target_failed = ?, target_message = ? where source_operation = ? and _index = ? and _type = ? and _id = ? and target_operation = 'n/a'";
926            PreparedStatement pstmt = connection.prepareStatement(sql);
927            for (BulkItemResponse resp : response) {
928                pstmt.setTimestamp(1, new Timestamp(new java.util.Date().getTime()));
929                pstmt.setString(2, resp.opType());
930                pstmt.setBoolean(3, resp.failed());
931                pstmt.setString(4, resp.failureMessage());
932                pstmt.setString(5, resp.opType());
933                pstmt.setString(6, resp.index());
934                pstmt.setString(7, resp.type());
935                pstmt.setString(8, resp.id());
936                pstmt.executeUpdate();
937            }
938            pstmt.close();
939            connection.commit();
940        } catch (SQLException ex) {
941            throw new IOException(ex);
942        }

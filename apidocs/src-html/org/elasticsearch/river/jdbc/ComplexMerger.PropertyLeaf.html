014/**
015 * Merge the results of the database. Algo is based on specified fieldnames.
016 * Differents sperators are authorized like . and [] to specify list of subfield
017  */
018public class ComplexMerger implements RowListener{
019
020    private ESLogger logger = ESLoggerFactory.getLogger("ComplexMerger");
021    private PropertyRoot root;
022
023
024    public ComplexMerger(){
025        this.root = createRoot();
026    }
027
028    public PropertyRoot createRoot(){
029        PropertyRoot root = new PropertyRoot();
030        return root;
031    }
032
033    public PropertyRoot getRoot(){
034        return this.root;
035    }
036
037    public void reset(){
038        this.root = new PropertyRoot();
039    }

041    public PropertyRoot run(String key,String value)throws Exception{
042        PropertyRoot root = new PropertyRoot();
043        merge(root, key, value);
044        return root;
045    }

048    @Override
049    public void row(String operation, String index, String type, String id, List<String> keys, List<Object> values) throws IOException {
050        if(keys == null || values == null || keys.size()!=values.size()){
051            throw new RuntimeException("Impossible");
052        }
053
054        PropertyRoot root = new PropertyRoot("Root");
055        for(int i = 0 ; i < keys.size() ; i++){
056            try{
057                merge(root,keys.get(i),values.get(i));
058            }catch(Exception e){
059                logger.error("Merge probleme",e);
060            }
061        }
062    }
063
064    @Override
065    public void row(List<String> keys, List<Object> values) throws IOException {
066        if(keys == null || values == null || keys.size()!=values.size()){
067            throw new RuntimeException("Impossible");
068        }
069        for(int i = 0 ; i < keys.size() ; i++){
070            try{
071                merge(root,keys.get(i),values.get(i));
072            }catch(Exception e){
073                logger.error("Probleme merge",e);
074            }
075            /* To avoid to have two duplicate elements in list */
076            root.deleteDuplicate();
077        }
078    }

082    public void merge(PropertyRoot root,String key,Object value)throws Exception{
083        /* Simple case, with value or multi value, leaf */
084        if(!key.contains(".") && !key.contains("[")){
085            if(root.containsNode(key)){
086                PropertyNode node = root.getNode(key);
087                if(node.isRoot()){
088                    throw new Exception("Suxx");
089                }
090                node.addValue(value);
091            }
092            else{
093                root.putProperty(new PropertyLeaf(key,value));
094            }
095            return;
096        }
097        /* Case with a list of object */
098        if(key.indexOf("[") !=-1 && (key.indexOf("[") < key.indexOf(".") || key.indexOf(".") == -1)){
099            // Extract the content in []
100            String rootKey = key.substring(0,key.indexOf("["));
101            String subContent = key.substring(key.indexOf("[")+1,key.indexOf("]"));
102            // If not property have been created, we create it
103            PropertyListRoot list = null;
104            if(root.containsNode(rootKey)){
105                PropertyNode node = root.getNode(rootKey);
106                if(!node.isRoot()){
107                    throw new Exception("Bad configuration");
108                }
109                list = (PropertyListRoot)node;
110            }
111            else{
112                list = new PropertyListRoot(rootKey);
113            }
114            root.putProperty(list);
115            merge(list,subContent,value);
116            return;
117        }
118
119        /* Case with sub object and property */
120        if(key.contains(".")){
121            String rootKey = key.substring(0,key.indexOf("."));
122            String subKey = key.substring(key.indexOf(".")+1);
123            if(!root.containsNode(rootKey,key)){
124                PropertyRoot subRoot = new PropertyRoot(rootKey);
125                root.putProperty(subRoot);
126            }
127            merge((PropertyRoot)root.getNode(rootKey),subKey,value);
128            return;
129        }
130
131    }

133    /**
134     * Node structure of JSON representation
135     */
136    public abstract class PropertyNode {
137        String name;
138
139        public String getName() {
140            return name;
141        }
142        /**
143         * If the the node is the root
144         * @return
145         */
146        public boolean isRoot(){return false;}
147
148        /**
149         * To delete duplicate in the list of sub node
150         */
151        public void deleteDuplicate(){}
152
153        /**
154         * To set the value of a node
155         * @param value
156         */
157        public abstract void addValue(Object value);
158
159        /**
160         * Get a quick json representation of node
161         * @return
162         */
163        public abstract String toString();
164
165        /**
166         * Get a XContentBuilder (use to generate json request) from the hierarchical property node
167         * @param builder   The builder in put informations
168         * @param writeTitle    If we write the title into node
169         * @return  The XContentBuilder representing the PropertyNode
170         * @throws IOException  When jsonBuilder throw IOException
171         */
172        public abstract XContentBuilder getXBuilder(XContentBuilder builder,boolean writeTitle)throws IOException;
173    }

175    /**
176     * Root representation of a tree. Contains some property
177     */
178    public class PropertyRoot extends PropertyNode{
179        private Map<String,PropertyNode> properties = new HashMap<String,PropertyNode>();
180        private String id;
181        private String operation;
182        public PropertyRoot(){}
183
184        public PropertyRoot(String name){
185            this.name = name;
186        }
187
188        /**
189         * Put a new property in the root
190         * @param propertyNode
191         */
192        public void putProperty(PropertyNode propertyNode){
193            properties.put(propertyNode.getName(),propertyNode);
194        }
195
196        public boolean isRoot(){return true;}
197
198
199        public PropertyNode getNode(String name){
200            return properties.get(name);
201        }
202        public boolean containsNode(String name,String completeKey){
203            return containsNode(name);
204        }
205
206        public boolean containsNode(String name){
207            return properties.containsKey(name);
208        }
209
210        public void addValue(Object value){
211            throw new RuntimeException("Impossible");
212        }
213
214        public String getOperation() {
215            return operation;
216        }
217
218        public void setOperation(String operation) {
219            this.operation = operation;
220        }
221
222        public String getId() {
223            return id;
224        }
225
226        public void setId(String id) {
227            this.id = id;
228        }
229
230        @Override
231        public void deleteDuplicate() {
232            for(PropertyNode p : properties.values()){
233                p.deleteDuplicate();
234            }
235        }
236
237        public String toString(){
238            StringBuilder builder = new StringBuilder();
239            builder.append("").append("{");
240            int pos = 0;
241            for(String s : properties.keySet()){
242                builder.append((pos++>0)?",":"").append("\"").append(s).append("\" : ").append(properties.get(s).toString());
243            }
244            builder.append("}");
245            return builder.toString();
246        }
247
248        public XContentBuilder getXBuilder()throws IOException{
249            return getXBuilder(null,false);
250        }
251
252        public XContentBuilder getXBuilder(XContentBuilder builder,boolean writeTitle)throws IOException{
253            if(builder == null){
254                builder = jsonBuilder();
255            }
256            if(writeTitle){
257                builder.startObject(this.name);
258            }else{
259                builder.startObject();
260            }
261            for(String s : properties.keySet()){
262                properties.get(s).getXBuilder(builder,true);
263            }
264            builder.endObject();
265            return builder;
266        }
267
268        @Override
269        public boolean equals(Object o) {
270            if (this == o) return true;
271            if (o == null || getClass() != o.getClass()) return false;
272
273            PropertyRoot that = (PropertyRoot) o;
274            if(properties.size() == 0 && that.properties.size() == 0){return true;}
275            if(properties.size()!= that.properties.size()){return false;}
276
277            for(String key : this.properties.keySet()){
278                if(!that.properties.containsKey(key)){return false;}
279                if(!this.properties.get(key).equals(that.properties.get(key))){return false;}
280            }
281            
282            return true;
283        }
284    }

287    /**
288     * Can stock many properties.
289     * Use to lead [] structure
290     * Use an internal map to know when to create a new element (with complete path)
291     */
292    public class PropertyListRoot extends PropertyRoot{
293        private LinkedList<PropertyRoot> properties = new LinkedList<PropertyRoot>();
294        /* use to know if a node has already been add to the properties. Based on the complete key */
295        private List<String> keyList = new ArrayList<String>();
296
297        public PropertyListRoot(String name){
298            this.name = name;
299        }
300
301        public boolean isRoot(){return true;}
302
303        /**
304         * Return a node which doesn't have value
305         * If it doesn't exist or already contains the name, add a new element a the end of list
306         * @param name
307         * @return
308         */
309        public PropertyNode getNode(String name){
310            PropertyRoot root = getLast();
311            if(root == null || root.containsNode(name,null)){
312                root = new PropertyRoot("");
313                properties.addLast(root);
314            }
315            return root;
316        }
317
318        /**
319         * Verify if last element is duplicate and delete it if yes
320         */
321        public void deleteDuplicate(){
322            if(properties.size()>0){
323                // We check the previous root not equals to an another element in the list (avoid duplicate)
324                // If last equals to another, delete it
325                PropertyRoot previousRoot = properties.getLast();
326                for(int i = 0 ; i < properties.size() -1 ; i++){    // evict the last element which is the current
327                    if(properties.get(i).equals(previousRoot)){
328                        properties.removeLast();
329                        break;
330                    }
331                }
332            }
333        }
334
335        /**
336         * Ajoute la propriete dans le dernier PropertyRoot. Si la propriete existe deja, recree un root et l'ajoute a la fin
337         * @param propertyNode
338         */
339        @Override
340        public void putProperty(PropertyNode propertyNode) {
341            PropertyRoot root = getLast();
342            if(root == null || root.containsNode(propertyNode.getName(),null)){
343                deleteDuplicate();  // not mandatory
344                root = new PropertyRoot("");
345                properties.addLast(root);
346            }
347            root.putProperty(propertyNode);
348        }
349
350        /**
351         * Special method which use the complete key to test if the node exist.
352         * If a node with this complete key exist, need to create a new Property root and define it as last
353         * @param name  Name of node
354         * @param completeKey Complete key to search in full index if exist
355         * @return
356         */
357        public boolean containsNode(String name,String completeKey){
358
359            if(completeKey!=null){
360                if(keyList.contains(completeKey)){
361                    // Case when we have already visit this node. Create a new Root
362                    PropertyRoot root = new PropertyRoot("");
363                    properties.addLast(root);
364                    keyList.removeAll(keyList);
365                }else{
366                    keyList.add(completeKey);
367                }
368                // To avoid creating root by calling method, all is lead here
369                return true;
370            }
371            return false;
372        }
373
374        public boolean containsNode(String name){
375            return false;
376        }
377
378        private PropertyRoot getLast(){
379            if(properties.size() == 0){
380                return null;
381            }
382            return properties.getLast();
383        }
384
385         public String toString(){
386            StringBuilder builder = new StringBuilder();
387            builder.append("[");
388            int pos = 0;
389            for(PropertyNode node : properties){
390                builder.append((pos++>0)?",":"").append(node.toString());
391            }
392            builder.append("]");
393            return builder.toString();
394        }
395
396        public XContentBuilder getXBuilder(XContentBuilder builder,boolean writeTitle)throws IOException{
397            if(builder == null){
398                builder = jsonBuilder();
399            }
400            builder.startArray(this.name);
401            for(PropertyNode node : properties){
402                node.getXBuilder(builder,false);
403            }
404            builder.endArray();
405
406            return builder;
407
408        }
409
410        @Override
411        public boolean equals(Object o) {
412            if (this == o) return true;
413            if (o == null || getClass() != o.getClass()) return false;
414
415            PropertyListRoot that = (PropertyListRoot) o;
416            if(properties.size() == 0 && that.properties.size() == 0){return true;}
417            if(properties.size()!= that.properties.size()){return false;}
418
419
420            for(int i = 0 ; i < this.properties.size() ; i++){
421                if(!this.properties.get(i).equals(that.properties.get(i))){return false;}
422            }
423
424            return true;
425        }
426
427    }

430    /**
431     * Leaf representation, contains a single value
432     * Can stock many values for a same key. The values a stock in a list
433     */
434    public class PropertyLeaf extends PropertyNode{
435        private List<Object> values = new ArrayList<Object>();
436
437        public PropertyLeaf(String key,Object value){
438            this.name = key;
439            values.add(value);
440        }
441
442        /**
443         * Set the value of the leaf. If a value have already been set, the new is add in the list (no replace)
444         * @param value
445         */
446        public void addValue(Object value){
447            if(!values.contains(value)){
448                values.add(value);
449            }
450        }
451
452        /**
453         * Return the differents values of the leaf
454         * @return
455         */
456        public List<Object> getValues() {
457            return values;
458        }
459
460        public String toString(){
461            StringBuilder builder = new StringBuilder();
462            int pos = 0;
463            for(Object str : values){
464                builder.append((pos++ > 0) ? "," : "").append("\"").append(str.toString()).append("\"");
465            }
466            if(values.size() > 1){
467                return "[" + builder.toString() + "]";
468            }
469            return builder.toString();
470        }
471
472        public XContentBuilder getXBuilder(XContentBuilder builder,boolean writeTitle)throws IOException{
473            builder.field(this.name);
474            if(values.size()>1){
475                builder.startArray();
476            }
477            for(Object str : values){
478                builder.value(str);
479            }
480            if(values.size()>1){
481                builder.endArray();
482            }
483            return builder;
484        }
485
486        @Override
487        public boolean equals(Object o) {
488            if (this == o) return true;
489            if (o == null || getClass() != o.getClass()) return false;
490
491            PropertyLeaf that = (PropertyLeaf) o;
492
493            if(that.getValues().size() == 0 && this.values.size() == 0){return true;}
494            if(that.getValues().size()!=this.values.size()){return false;}
495
496            for(int i = 0 ; i < this.values.size() ; i++){
497                if(!this.values.get(i).equals(that.getValues().get(i))){return false;}
498            }
499            return true;
500        }
501    }

087    @Inject
088    public JDBCRiver(RiverName riverName, RiverSettings settings,
089                     @RiverIndexName String riverIndexName, Client client) {
090        super(riverName, settings);
091        this.riverIndexName = riverIndexName;
092        this.client = client;
093        if (settings.settings().containsKey("jdbc")) {
094            Map<String, Object> jdbcSettings = (Map<String, Object>) settings.settings().get("jdbc");
095            poll = XContentMapValues.nodeTimeValue(jdbcSettings.get("poll"), TimeValue.timeValueMinutes(60));
096            url = XContentMapValues.nodeStringValue(jdbcSettings.get("url"), null);
097            driver = XContentMapValues.nodeStringValue(jdbcSettings.get("driver"), null);
098            user = XContentMapValues.nodeStringValue(jdbcSettings.get("user"), null);
099            password = XContentMapValues.nodeStringValue(jdbcSettings.get("password"), null);
100            sql = XContentMapValues.nodeStringValue(jdbcSettings.get("sql"), null);
101            strategy = XContentMapValues.nodeStringValue(jdbcSettings.get("strategy"), null);
102            fetchsize = XContentMapValues.nodeIntegerValue(jdbcSettings.get("fetchsize"), 0);
103            params = XContentMapValues.extractRawValues("params", jdbcSettings);
104            mapping = XContentMapValues.extractRawValues("mapping", jdbcSettings);
105            rivertable = XContentMapValues.nodeBooleanValue(jdbcSettings.get("rivertable"), false);
106            interval = XContentMapValues.nodeTimeValue(jdbcSettings.get("interval"), TimeValue.timeValueMinutes(60));
107            versioning = XContentMapValues.nodeBooleanValue(jdbcSettings.get("versioning"), true);
108            rounding = XContentMapValues.nodeStringValue(jdbcSettings.get("rounding"), null);
109            scale = XContentMapValues.nodeIntegerValue(jdbcSettings.get("scale"), 0);
110        } else {
111            poll = TimeValue.timeValueMinutes(60);
112            url = null;
113            driver = null;
114            user = null;
115            password = null;
116            sql = null;
117            strategy = null;
118            fetchsize = 0;
119            params = null;
120            mapping = null;
121            rivertable = false;
122            interval = TimeValue.timeValueMinutes(60);
123            versioning = true;
124            rounding = null;
125            scale = 0;
126        }
127        if (settings.settings().containsKey("index")) {
128            Map<String, Object> indexSettings = (Map<String, Object>) settings.settings().get("index");
129            indexName = XContentMapValues.nodeStringValue(indexSettings.get("index"), "jdbc");
130            typeName = XContentMapValues.nodeStringValue(indexSettings.get("type"), "jdbc");
131            bulkSize = XContentMapValues.nodeIntegerValue(indexSettings.get("bulk_size"), 100);
132            maxBulkRequests = XContentMapValues.nodeIntegerValue(indexSettings.get("max_bulk_requests"), 30);
133            if (indexSettings.containsKey("bulk_timeout")) {
134                bulkTimeout = TimeValue.parseTimeValue(XContentMapValues.nodeStringValue(indexSettings.get("bulk_timeout"), "60s"), TimeValue.timeValueMillis(60000));
135            } else {
136                bulkTimeout = TimeValue.timeValueMillis(60000);
137            }
138        } else {
139            indexName = "jdbc";
140            typeName = "jdbc";
141            bulkSize = 100;
142            maxBulkRequests = 30;
143            bulkTimeout = TimeValue.timeValueMillis(60000);
144        }
145        service = new SQLService(logger);
146        operation = new BulkOperation(client, logger).setIndex(indexName).setType(typeName)
147                .setBulkSize(bulkSize).setMaxActiveRequests(maxBulkRequests)
148                .setMillisBeforeContinue(bulkTimeout.millis()).setMillisBeforeContinue(bulkTimeout.millis()).setAcknowledge(riverName.getName(), rivertable ? service : null);
149
150        // Strategy
151        riverStrategy = new JDBCConnector();
152        if(strategy!=null && strategy.equals("timebasis")){
153            riverStrategy = new JDBCConnectorTimebasis();
154        }
155        if(strategy!=null && strategy.equals("rivertable")){
156            riverStrategy = new JDBCRiverTableConnector();
157        }
158
159    }

161    @Override
162    public void start() {
163        logger.info("starting JDBC connector: URL [{}], driver [{}], sql [{}], river table [{}], indexing to [{}]/[{}], poll [{}]",
164                url, driver, sql, rivertable, indexName, typeName, poll);
165        try {
166            client.admin().indices().prepareCreate(indexName).execute().actionGet();
167            creationDate = new Date();
168        } catch (Exception e) {
169            if (ExceptionsHelper.unwrapCause(e) instanceof IndexAlreadyExistsException) {
170                creationDate = null;
171                // that's fine
172            } else if (ExceptionsHelper.unwrapCause(e) instanceof ClusterBlockException) {
173                // ok, not recovered yet..., lets start indexing and hope we recover by the first bulk
174            } else {
175                logger.warn("failed to create index [{}], disabling river...", e, indexName);
176                return;
177            }
178        }
179        thread = EsExecutors.daemonThreadFactory(settings.globalSettings(), "JDBC connector").newThread(riverStrategy);
180        thread.start();
181    }
182
183    @Override
184    public void close() {
185        if (closed) {
186            return;
187        }
188        logger.info("closing JDBC river");
189        thread.interrupt();
190        closed = true;
191    }

193    private class JDBCConnector implements Runnable {
194
195        @Override
196        public void run() {
197            while (true) {
198                try {
199                    Number version;
200                    String digest;
201                    // read state from _custom
202                    client.admin().indices().prepareRefresh(riverIndexName).execute().actionGet();
203                    GetResponse get = client.prepareGet(riverIndexName, riverName().name(), ID_INFO_RIVER_INDEX).execute().actionGet();
204                    if (creationDate != null || !get.exists()) {
205                        version = 1L;
206                        digest = null;
207                    } else {
208                        Map<String, Object> jdbcState = (Map<String, Object>) get.sourceAsMap().get("jdbc");
209                        if (jdbcState != null) {
210                            version = (Number) jdbcState.get("version");
211                            version = version.longValue() + 1; // increase to next version
212                            digest = (String) jdbcState.get("digest");
213                        } else {
214                            throw new IOException("can't retrieve previously persisted state from " + riverIndexName + "/" + riverName().name());
215                        }
216                    }
217                    Connection connection = service.getConnection(driver, url, user, password, true);
218                    PreparedStatement statement = service.prepareStatement(connection, sql);
219                    service.bind(statement, params);
220                    ResultSet results = service.execute(statement, fetchsize);
221                    Merger merger = new Merger(operation, version.longValue());
222                    long rows = 0L;
223                    while (service.nextRow(results, merger)) {
224                        rows++;
225                    }
226                    merger.close();
227                    service.close(results);
228                    service.close(statement);
229                    service.close(connection);
230                    logger.info("got " + rows + " rows for version " + version.longValue() + ", digest = " + merger.getDigest());
231                    // this flush is required before house keeping starts
232                    operation.flush();
233                    // save state to _custom
234                    XContentBuilder builder = jsonBuilder();
235                    builder.startObject().startObject("jdbc");
236                    if (creationDate != null) {
237                        builder.field("created", creationDate);
238                    }
239                    builder.field("version", version.longValue());
240                    builder.field("digest", merger.getDigest());
241                    builder.endObject().endObject();
242                    client.prepareBulk().add(indexRequest(riverIndexName).type(riverName.name()).id(ID_INFO_RIVER_INDEX).source(builder)).execute().actionGet();
243                    // house keeping if data has changed
244                    if (digest != null && !merger.getDigest().equals(digest)) {
245                        housekeeper(version.longValue());
246                        // perform outstanding housekeeper bulk requests
247                        operation.flush();
248                    }
249                    delay("next run");
250                } catch (Exception e) {
251                    logger.error(e.getMessage(), e, (Object) null);
252                    closed = true;
253                }
254                if (closed) {
255                    return;
256                }
257            }
258        }
259        
260        private void housekeeper(long version) throws IOException {
261            logger.info("housekeeping for version " + version);
262            client.admin().indices().prepareRefresh(indexName).execute().actionGet();
263            SearchResponse response = client.prepareSearch().setIndices(indexName).setTypes(typeName).setSearchType(SearchType.SCAN).setScroll(TimeValue.timeValueMinutes(10)).setSize(bulkSize).setVersion(true).setQuery(matchAllQuery()).execute().actionGet();
264            if (response.timedOut()) {
265                logger.error("housekeeper scan query timeout");
266                return;
267            }
268            if (response.failedShards() > 0) {
269                logger.error("housekeeper failed shards in scan response: {0}", response.failedShards());
270                return;
271            }
272            String scrollId = response.getScrollId();
273            if (scrollId == null) {
274                logger.error("housekeeper failed, no scroll ID");
275                return;
276            }
277            boolean done = false;
278            // scroll
279            long deleted = 0L;
280            long t0 = System.currentTimeMillis();
281            do {
282                response = client.prepareSearchScroll(response.getScrollId()).setScroll(TimeValue.timeValueMinutes(10)).execute().actionGet();
283                if (response.timedOut()) {
284                    logger.error("housekeeper scroll query timeout");
285                    done = true;
286                } else if (response.failedShards() > 0) {
287                    logger.error("housekeeper failed shards in scroll response: {}", response.failedShards());
288                    done = true;
289                } else {
290                    // terminate scrolling?
291                    if (response.hits() == null) {
292                        done = true;
293                    } else {
294                        for (SearchHit hit : response.getHits().getHits()) {
295                            // delete all documents with lower version
296                            if (hit.getVersion() < version) {
297                                operation.delete(hit.getIndex(), hit.getType(), hit.getId());
298                                deleted++;
299                            }
300                        }
301                        scrollId = response.getScrollId();
302                    }
303                }
304                if (scrollId != null) {
305                    done = true;
306                }
307            } while (!done);
308            long t1 = System.currentTimeMillis();
309            logger.info("housekeeper ready, {} documents deleted, took {} ms", deleted, t1 - t0);
310        }
311    }

313    private class JDBCConnectorTimebasis implements Runnable {
314        String previousLastModificationDate = null;
315
316        @Override
317        public void run() {
318            // Use to avoid to ignore last insert object when resultset are done. The second pass, use >= to get results created with the same modification date, but after the selection.
319            // If last date are same, strict greater than to avoid index the same objects
320           while (true) {
321                try {
322                    /* Test of differents parameters.*/
323                    /* Mandatory : _id, _modificationDate must be in select sql request */
324                    if(!sql.contains("_id")){
325                        // Error
326                        throw new Exception("Field \"_id\" is mandatory in select clause");
327                    }
328
329                    if(!sql.contains(FIELD_MODIFICATION_DATE)){
330                        // Error
331                        throw new Exception("Field \"" + FIELD_MODIFICATION_DATE + "\" is mandatory in select clause");
332                    }
333                    if(mapping == null){
334                        throw new Exception("Mapping is mandatory");
335                    }
336
337                    // subselect : sql * from (select a,b,c from aa,bb,cc ...) where date >= aliasDateField order _id asc
338                    String requestSQL = "select * from (" + sql + ") ";
339
340                    String lastModificationDate = getPreviousDateModification();
341
342                    /* Add modification date filter */
343                    if(lastModificationDate!=null){
344                        requestSQL+= " where \"" + FIELD_MODIFICATION_DATE + "\""
345                                + (lastModificationDate.equals(previousLastModificationDate)?">":">=")
346                                + "?";
347                        previousLastModificationDate = lastModificationDate;
348                    }
349
350                    /* Add the order instruction : id and modification date */
351                    //requestSQL += " order by \"" + FIELD_MODIFICATION_DATE + "\" asc, \"_id\" asc";
352                    requestSQL += " order by \"_id\" asc";
353
354                    logger.info("Requete SQL : " + requestSQL);
355
356                    String indexOperation = sql.contains("_operation") ? null : "index";
357
358                    Connection connection = service.getConnection(driver, url, user, password, true);
359                    PreparedStatement statement = service.prepareStatement(connection, requestSQL);
360                    service.bind(statement, params);
361
362                    if(lastModificationDate!=null){
363                        List<Object> list = new ArrayList<Object>();
364                        list.add(Timestamp.valueOf(lastModificationDate));
365                        service.bind(statement, list);
366                    }
367
368                    lastModificationDate = service.treat(statement, fetchsize,indexOperation,operation,mapping);
369
370                    // If no results to index, the lastmodificationdate is null, get the previous date
371                    if(lastModificationDate == null){
372                        lastModificationDate = previousLastModificationDate;
373                    }
374                    service.close(statement);
375                    service.close(connection);
376
377                    saveContexteInIndex(lastModificationDate,"OK");
378
379                    if(delay){
380                        delay("next run");
381                    }
382                    else{
383                        closed = true;
384                    }
385                } catch (Exception e) {
386                    logger.error(e.getMessage(), e, (Object) null);
387                    closed = true;
388                    try{
389                        saveContexteInIndex(null,"KO");
390                    }catch(Exception ex){
391                        logger.error("Impossible te define status in index");
392                    }
393                }
394                if (closed) {
395                    return;
396                }
397            }
398        }
399
400        /**
401         * The modification date of the last index document
402         * @return
403         */
404        private String getPreviousDateModification()throws Exception{
405            client.admin().indices().prepareRefresh(riverIndexName).execute().actionGet();
406            GetResponse get = client.prepareGet(riverIndexName, riverName().name(), ID_INFO_RIVER_INDEX).execute().actionGet();
407            if (!get.exists()) {
408                return null;
409            } else {
410                Map<String, Object> jdbcState = (Map<String, Object>) get.sourceAsMap().get("jdbc");
411                if (jdbcState != null) {
412                    return (String)jdbcState.get("lastDateModification");
413                } else {
414                    throw new IOException("can't retrieve previously persisted state from " + riverIndexName + "/" + riverName().name());
415                }
416            }
417        }
418
419        private void saveContexteInIndex(String lastDateModification,String statut)throws Exception{
420            logger.info("Last modification date : " + lastDateModification);
421            XContentBuilder builder = jsonBuilder();
422            builder.startObject().startObject("jdbc");
423            if (lastDateModification != null) {
424                builder.field("lastDateModification", lastDateModification);
425            }
426            builder.field("statut",statut);
427            builder.field("lastExecution", Calendar.getInstance().getTime());
428            builder.endObject().endObject();
429            client.prepareIndex(riverIndexName,riverName.name(),ID_INFO_RIVER_INDEX).setSource(builder).execute().actionGet();
430        }

433    private class JDBCRiverTableConnector implements Runnable {
434
435        private String[] optypes = new String[]{"create", "index", "delete"};
436
437        @Override
438        public void run() {
439            while (true) {
440                for (String optype : optypes) {
441                    try {
442                        Connection connection = service.getConnection(driver, url, user, password, false);
443                        PreparedStatement statement = service.prepareRiverTableStatement(connection, riverName.getName(), optype, interval.millis());
444                        ResultSet results = service.execute(statement, fetchsize);
445                        Merger merger = new Merger(operation);
446                        long rows = 0L;
447                        while (service.nextRiverTableRow(results, merger)) {
448                            rows++;
449                        }
450                        merger.close();
451                        service.close(results);
452                        service.close(statement);
453                        logger.info(optype + ": got " + rows + " rows");
454                        // this flush is required before next run
455                        operation.flush();
456                        service.close(connection);
457                    } catch (Exception e) {
458                        logger.error(e.getMessage(), e, (Object) null);
459                        closed = true;
460                    }
461                    if (closed) {
462                        return;
463                    }
464                }
465                delay("next run");
466            }
467        }
468    }

471    private void delay(String reason) {
472        if (poll.millis() > 0L) {
473            logger.info("{}, waiting {}, URL [{}] driver [{}] sql [{}] river table [{}]",
474                    reason, poll, url, driver, sql, rivertable);
475            try {
476                Thread.sleep(poll.millis());
477            } catch (InterruptedException e1) {
478            }
479        }
480    }
